Checking if the colors are too dark for a black font
#+begin_src python
def is_dark(color: str) -> bool:
    denom = 255
    color = int(color[1:],16)
    r = (color >> 16) & 0xff
    g = (color >> 8) & 0xff
    b = (color >> 0) & 0xff
    value = 0.212 * r + 0.701 * g + 0.087 * b
    return (value / denom) < 0.25
#+end_src
